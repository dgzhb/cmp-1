repo=sae
git remote add $repo https://git.sinacloud.com/cmpdemo
## from remote :1 to local branch 
git pull $repo sae:1

# 普通 git 用法（看手册）
git status
git add .
git status
git commit -m "...."

# 上传代码，比如从本地 branch\sae到原程 master:1 
git push $repo master:1
